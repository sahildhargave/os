<script>
    document.getElementById('registrationForm').addEventListener('submit', function(event) {
        var firstName = document.getElementById('firstName').value.trim();
        var lastName = document.getElementById('lastName').value.trim();
        var organization = document.getElementById('organization').value.trim();
        var hobbies = document.getElementById('hobbies').value.trim();

        var firstNameError = document.getElementById('firstNameError');
        var lastNameError = document.getElementById('lastNameError');
        var organizationError = document.getElementById('organizationError');

        var isValid = true;

        // Validation for first name
        if(firstName === '') {
            firstNameError.textContent = 'First name is required';
            isValid = false;
            setTimeout(function() {
                firstNameError.textContent = '';
            }, 2000); // Clear error message after 2 seconds
        } else {
            firstNameError.textContent = '';
        }

        // Validation for last name
        if(lastName === '') {
            lastNameError.textContent = 'Last name is required';
            isValid = false;
            setTimeout(function() {
                lastNameError.textContent = '';
            }, 2000); // Clear error message after 2 seconds
        } else {
            lastNameError.textContent = '';
        }

        // Validation for organization (accept only string)
        if(!isNaN(organization)) {
            organizationError.textContent = 'Organization must be a string';
            isValid = false;
            setTimeout(function() {
                organizationError.textContent = '';
            }, 2000); // Clear error message after 2 seconds
        } else {
            organizationError.textContent = '';
        }

        if(isValid) {
            event.preventDefault(); // Prevent form submission if validation fails

            // Display details
            document.getElementById('displayFirstName').textContent = firstName;
            document.getElementById('displayLastName').textContent = lastName;
            document.getElementById('displayOrganization').textContent = organization;
            document.getElementById('displayHobbies').textContent = hobbies;

            // Show details section
            document.getElementById('detailsSection').style.display = 'block';
        }
    });
</script>
